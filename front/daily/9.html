# JS第九天

# 1.昨天作业

document.getElementById("btn").onclick = function () {
  // 获取表格
  var tb = document.getElementById("tb");
  // 获取昵称框内的内容
  var name = document.getElementById("userName");
  // 获取文本域中的内容
  var text = document.getElementById("tt");
  // 创建一个tr
  var tr = document.createElement("tr");
  tb.appendChild(tr);
  // 创建两个td
  var td1 = document.createElement("td");
  // 把内容追加到td里面,把td追加到tr里
  tr.appendChild(td1);
  td1.innerHTML = name.value;
  // 第二个td
  var td2 = document.createElement("td");
  tr.appendChild(td2);
  td2.innerHTML = text.value;
  // 清空
  name.value = "";
  text.value = "";
}

# 2.面向对象思想

  • 面向对象思想: 如果我们要完成一个需求,首先要分析需求
  • 然后去找对应的对象,要找到对应的对象,首先得有对象,有对象的
  • 前提是有构造函数,通过构造函数创建对象,再调用对象中对应的
  • 方法完成需求,搞定了

# 3.Js语言是面向什么的

  • JS不是一门面向对象的语言,它是一门基于对象的语言
  • 那为什么我们还要去学习面向对象思想呢,因为这种思想
  • 更加符合人们的思考方式,利用这种思想编程可以简化我们的代码
  • 而且利于后期的维护

# 4.面向对象的特征

  • 面向对象三大特征: 封装、继承、多态、

# 5.Js封装

  • 封装:把一堆重复代码放到一个代码块中 -- 简化代码

# 6.Js继承

  • 继承:首先继承是一种关系,类(class)与类之间的关系,
  • 但是JS中没有类,JS通过构造函数来模拟类,通过原型来模拟继承

# 7.原形的作用

  • 原型作用:
  • 实现数据共享
  • 实现继承 -- 实现数据共享、简化代码

# 8.改变原型指向来实现继承

  • 缺点: 对象的属性不能修改
  • 解决办法:1、可以通过重新赋值的方式来修改,但是比较麻烦 2、不使用改变原型指向的方式
  • 直接借用父级的构造函数 通过call(当前对象,属性值,属性值.....);
  • （但是还存在一个问题: 父级的方法无法被继承）

# 9.组合继承

  • 改变原型指向 -- 实现方法的继承
  • 借用父级构造函数 -- 实现属性的继承

# 10.拷贝继承

  • 就是把一个对象的属性和方法拷贝到另一个对象
  • //就是把一个对象的属性和方法拷贝到另一个对象
var obj = {
  name:"张三",
  age:"12",
  sex:"女",
  sayHi:function () {
    console.log("你好啊");
  }
};
//浅拷贝
var obj2 = {};
for (var key in obj){
    obj2[key] = obj[key];
}
console.log(obj2.name);
console.log(obj2.age);
console.log(obj2.sex);
obj2.sayHi();
var obj = {
    name:"张三",
    age:"12",
    sex:"女",
    sayHi:function () {
        console.log("你好啊");
    }
};
function Obj(name,age,sex){
    this.name = name;
    this.age = age;
    this.sex = sex;
}
Obj.prototype.like = "蹦极";
Obj.prototype.eat = function () {
    console.log("榴莲炸臭豆腐");
};
// 因为prototype既是一个属性也是一个对象
// 所以它也能拷贝
// 其实并没有全部拷贝过来,因为prototype中有些东西是不让我们拷贝的
var obj2 = {};
for (var key in Obj.prototype){
    obj2[key] = Obj.prototype[key];
}
console.log(obj2.like);
obj2.eat();

# 11.设置一个随机出现的点

  • （后期可以用做贪吃蛇制作）
var l = document.querySelector(".little");//获取小div的值（最好是刷新出来自己创建一个，这样不会被绑死，符合面向对象）
var b = document.querySelector(".big");//获取大div的值
var sw = Math.ceil(Math.random()*(b.offsetWidth-l.offsetWidth));//设置一个随机数，限制范围是大div的宽度减去小div的宽度
var sw1 = Math.ceil(Math.random()*(b.offsetHeight-l.offsetHeight));//设置第二个随机数，限制范围是大div的高度减去小div的高度
l.style.left = sw+"px";//小div的定位left值为第一个随机数
l.style.top = sw1+"px";//小div的定位top值为第二个随机数
