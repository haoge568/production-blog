for (let i = 0; i < spans.length; i++) {
  // onclick事件是异步的,会被加载到任务队列
  spans[i].onclick = function () {
    for (var j = 0; j < spans.length; j++) {
      // 把span的样式清空

      // 把li的样式清空
      lis[j].removeAttribute("class");
    }

    // console.log(i);

拓展：Attribute
attributes：获取一个属性作为对象
getAttribute：获取某一个属性的值
object.getAttributes(attribute)
getAttribute方法不属于document对象，所以不能通过document对象获取，只能通过元素节点的调用。
例如document.getElementsByTagName("p")[0].getAttributes("title")
setAttribute：建立一个属性，并同时给属性捆绑一个值允许对属性节点做出修改
createAttribute：仅建立一个属性
removeAttribute：删除一个属性
getAttributeNode：获取一个节点作为对象
setAttributeNode：建立一个节点
removeAttributeNode：删除一个节点

# 2.节点的相关属性

  • 元素: 页面中所有的标签
  • 节点: 页面中所有的内容(标签,属性,文本(回车、空格、注释))
  • 根元素: html
  • 文档对象: document
  • 用之前所学的知识完成以下需求 -- 不方便
  • 节点属性:
  • nodeType: 节点类型 1 -- 元素 2 -- 属性 3 -- 文本
  • nodeName: 节点名字 标签--大写的标签名 属性--小写的属性名 文本--#text
  • nodeValue:节点值 标签--null 属性值--小写属性值 文本--文本内容

# 6.节点的兼容代码

  • 两个封装函数可以应对各种不同浏览器的兼容
  • // 获取任何父级元素的第一个子元素
function getFirstByParent(ele) {
  if (ele.firstElementChild) {// !undefined --> true
    // 如果能进这个判断  说明if里面的结果不是undefined
    // 即浏览器支持该属性
    return ele.firstElementChild;
  } else {
    // return ele.firstChild;
    // 如果有一个浏览器,既不支持firstElementChild
    // 通过firstChild返回的又是第一个节点而不是元素
    var node = ele.firstChild;
    while (node && node.nodeType != 1) {
        node = ele.nextSibling;
    }
    return node; // 百分百标签
  }
}

// 获取任何父级元素的最后子元素
function getLastByParent(ele) {
  if (ele.lastElementChild) {// !undefined --> true
    // 如果能进这个判断  说明if里面的结果不是undefined
    // 即浏览器支持该属性
    return ele.lastElementChild;
  } else {
    // return ele.firstChild;
    // 如果有一个浏览器,既不支持firstElementChild
    // 通过firstChild返回的又是第一个节点而不是元素
    var node = ele.lastChild;
    while (node && node.nodeType != 1) {
      node = ele.previousSibling;
    }
    return node; // 百分百标签
  }
}

# 7.练习：节点实现隔行变色

var lis = my$("uu").childNodes;
var count = 0;
for (var i = 0; i < lis.length; i++) {
  if (lis[i].nodeType == 1 && lis[i].nodeName == "LI") {
    lis[i].style.backgroundColor = count % 2 == 0 ? "red" : "yellow";
    count++;
  }
}

# 8.字符串动态创建列表

var date = ["九阳神功", "降龙十八掌", "如来神掌", "极乐神功", "凌波微步"];
my$("btn").onclick = function () {
  var str = "<ul style='list-style: none;cursor: pointer'>";
  for (var i = 0; i < date.length; i++) {
      str += "<li>" + date[i] + "</li>";
  }
  str += "</ul>";
  my$("dv").innerHTML = str;
  var lis = document.getElementsByTagName("li");
  for (var j = 0; j < lis.length; j++) {
    // lis[j].onmouseover = function(){
    //
    // };
    // 为什么不按上面的方式写呢

    lis[j].onmouseover = mouseoverHandler;
    lis[j].onmouseout = mouseoutHandler;
  }
  function mouseoverHandler() {
    this.style.backgroundColor = "yellow";
  }
  function mouseoutHandler() {
    this.style.backgroundColor = "";
  }
  // 如果在for循环里面添加事件,把事件定义在for循环外面
  // 循环里面直接赋值,可以提高性能
}
